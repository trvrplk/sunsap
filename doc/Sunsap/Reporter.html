def initialize(reportername)
  @queue = []

  # Public: Send a message to the queue
  #
  # m - message to send
  #
  # Example:
  #
  #   reporter.send_test_message "Success!"
  #   # => true
  self.class.send :define_method, "send_#{reportername}_message" do |m|
    @queue.push(m)
    true
  end

  # Public: View the queue
  #
  # Example:
  #
  #   reporter.queue
  #   # => ["Success!"]
  def queue
    @queue
  end
end
